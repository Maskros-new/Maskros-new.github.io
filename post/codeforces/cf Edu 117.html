void solve() {
    int n,a,b; cin>>n>>a>>b;
    memset(vis,0,sizeof(vis));
    memset(num, 0, sizeof(num));
    vis[a]=1;
    vis[b]=1;
    num[n/2+1]=b;
    num[1]=a;
    int tmp=1;
    bool find;
    rep(i,n/2+2,n+1){
        find=false;
        rep(j,tmp,b){
            if(!vis[j]){
                vis[j]=1;
                num[i]=j;
                tmp=j+1;
                find=1;
                break;
            }
        }

                find=1;

C_Chat_Ban

题意：

输入k,x ; 你有2k-1条信息要发，信息的长度为1,2,3…k-1,k,k-1…1，在发出当前信息时，你已经发送的总长度必须小于x，问你最多能发多少条

思路：

二分答案即可

#include <bits/stdc++.h>

#define gcd(a, b) __gcd(a, b)
#define INF 0x3f3f3f3f3f
#define eps 1e-6
#define PI acos(-1.0)
#define pb push_back
#define mp make_pair
#define fst first
#define sec second
#define eif else if
#define en putchar('\n')
#define rep(i, x, y) for (int i = x; i < y; i++)
#define red(i, x, y) for (int i = x - 1; i >= y; i--)
#define mem(a, x) memset(a, x, sizeof(a))
#define IOS cin.tie(0), ios::sync_with_stdio(false)
#define maxn 400005
#define co(x) cout << x << " ";
typedef long long ll;
#define pll pair<ll, ll>
using namespace std;
ll cal(ll begin, ll end){
    ll ret;
    if((begin+end)%2==0){
        ret=(begin+end)/2*(end-begin+1);
    }else{
        ret=(end-begin+1)/2*(begin+end);
    }
    return ret;
}
void solve() {
    ll k,x; cin>>k>>x;
    ll ans=0;
    ll test=cal(1,k);
    if(test==x){
        cout<<k<<endl;
        return;
    }eif(test<x){
        ans+=k;
        x-=test;
        test=cal(1,k-1);
        if(test<=x){ cout<<2*k-1<<endl; return;}
        else{
            ll l=1,r=k;
            ll mid;
            ll tmp=0;
            while(l<=r){
                mid=(l+r)>>1;
                test=cal(mid,k-1);
                if(test==x){
                    tmp=k-mid;
                    break;
                }eif(test<x){
                    tmp=k-mid+1;
                    r=mid-1;
                }else{
                    l=mid+1;
                }
            }
            ans+=tmp;
            cout<<ans<<endl;
        }
    }else{
        ll l=1,r=k;
        ll mid;
        while(l<=r){
            mid=(l+r)>>1;
            test=cal(1,mid);
            if(test==x){
                ans=mid;
                break;
            }eif(test<x){
                ans=mid+1;
                l=mid+1;
            }else{
                r=mid-1;
            }
        }
        cout<<ans<<endl;
    }

}
int main() {
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
}

D_X_Magic_Pair (gcd)

题意：

给你 a, b, x，每次操作可以将a或b替换为 |a-b|, 询问能否将a或b进行任意次操作后替换为 x

思路：

gcd变种，每次判断一下 a%b 是否等于 x%b，即 (a-x)%b==0 ，如果相等即为YES，否则即为NO

#include <bits/stdc++.h>

#define INF 0x3f3f3f3f3f
#define eps 1e-6
#define PI acos(-1.0)
#define pb push_back
#define mp make_pair
#define fst first
#define sec second
#define eif else if
#define en putchar('\n')
#define rep(i, x, y) for (int i = x; i < y; i++)
#define red(i, x, y) for (int i = x - 1; i >= y; i--)
#define mem(a, x) memset(a, x, sizeof(a))
#define IOS cin.tie(0), ios::sync_with_stdio(false)
#define maxn 400005
#define co(x) cout << x << " ";
typedef long long ll;
#define pll pair<ll, ll>
using namespace std;
ll a, b, x;
bool gcd(ll a, ll b){
    if(a==x||b==x) return 1;
    if(b==0) return 0;
    if(a>x&&(a-x)%b==0) return 1;    
    if(b>x&&(b-x)%a==0) return 1;
    return gcd(b,a%b);
}
void solve() {
    cin>>a>>b>>x;
    if(x>a&&x>b){ cout<<"NO"<<endl; return; }
    if(x==a||x==b){ cout<<"YES"<<endl; return;}
    ll tmpa=(a>b)?a:b;
    ll tmpb=(b<a)?b:a;
    if(gcd(tmpa,tmpb)) puts("YES");
    else puts("NO");
}
int main() {
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
}
