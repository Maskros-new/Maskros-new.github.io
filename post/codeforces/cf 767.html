A_Download More RAM

水题，排序即可

pair<int,int> pr[105];
bool cmp(pair<int,int> a, pair<int,int> b){
    return a.fst<b.fst;
}
void solve() {
    int n,k; cin>>n>>k;
    rep(i,0,n){
        cin>>pr[i].fst;
    }
    rep(i, 0, n) {
        cin >> pr[i].sec;
    }
    sort(pr,pr+n,cmp);
    ll tmp=k;
    rep(i,0,n){
        if(tmp>=pr[i].fst) tmp+=pr[i].sec;
        else break;
    }
    cout<<tmp<<endl;
}
int main() {
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
}

B_GCD Arrays

题意：

对一个 [l,r] 区间里的数，最多操作k次，每次取两个数，将他们的乘积替代其中一个数放入序列中，问能不能使序列的GCD不为1，能即YES，不能即NO

思路：

粗略的想了一下，乘积之后肯定包含两个原数作为因子，没啥区别，考虑连续区间、奇偶性后，猜测结论：每次操作使奇数*偶数=偶数，将乘积代替奇数，至多k次操作能否使得序列全变为偶数(GCD=2)，注意特判

写的比较仓促，所以比较丑，可以简化= =

void solve() {
    int l,r,k; cin>>l>>r>>k;
    int ou=0,ji=0;
    if((r-l+1)&1){
        if(l&1){
            ji++;
        }else ou++;
        ji+=(r-l)>>1;
        ou+=(r-l)>>1;
    }else{
        ou=(r-l+1)>>1; ji=ou;
    }
    if(ou==0){
        if(l!=1) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }else{
        if(ji<=k) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
}
int main() {
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
}

C_Meximum Array

Mex：一个序列中不存在的最小的非负整数

题意：

给你一个长为n的序列a，每次从a数组的首地址开始选取一段，求mex后将值存入b，随后在a中删去该段。问最终求得的字典序最大的b序列是什么样子的

思路：

模拟（队友用主席树艹过了xD）一个map<int,int> m表示a[i]出现的次数，每次求最大mex的时候即利用该map，删除序列即对map进行操作即可；set<int> S表示从序列开始所有小于mex的数，如果 S.size()<=mex，继续拓展序列的长度就不是最优解，故该位置即作为选取序列的结尾

卡了很久，刚开始想了个n2模拟，结果T了，比较傻逼，后来优化一下就过了

int a[maxn];
map<int, int> m;
int check(int start) {
    rep(i, start, maxn) {
        if (m[i] == 0) {
            return i;
        }
    }
    return 0;
}
vector<int> b;
set<int> S;
void solve() {
    int n;
    cin >> n;
    int mex = 0;
    m.clear();
    b.clear();
    rep(i, 0, n) {
        cin >> a[i];
        m[a[i]]++;
        if (m[mex] > 0) {
            mex = check(mex);
        }
    }
    S.clear();
    rep(i, 0, n) {
        if (a[i] < mex)
            S.insert(a[i]);
        m[a[i]]--;
        if (S.size() >= mex) {
            b.pb(mex);
            mex = check(0);
            S.clear();
        }
    }
    cout << b.size() << endl;
    rep(i, 0, b.size()) cout << b[i] << " ";
    cout << endl;
}
int main() {
    IOS;
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
}

D_Peculiar Movie Preferences

题意：

给你一堆长<=3的字符串，问能不能通过删掉任意个数的字符串，剩下的字符串按顺序拼接起来为回文串

思路：

沙比题，回文串首尾必定一样，所以如果可以的话最多找两个串即可组成回文串，总长度<=6。首先特判单个串，如果单个回文即为YES；再者分别考虑 2+2/3+3, 2+3/3+2 的情况，特别是后面那一种，这里通过map优化，对字符串做reverse处理，针对不同情况判断即可 TuT

map<string,int> mp1,mp2;
void solve() {
    int n;
    cin >> n;
    bool can = 0;
    string s,ss;
    mp1.clear(); mp2.clear();
    rep(i, 0, n) {
        cin >> s;
        if(!can){
            ss = s;
            reverse(ss.begin(), ss.end());
            if (s.size() == 1) can = 1;
            eif(s[0] == s[s.size() - 1]) can = 1;
            else{
                if(mp1[ss]==1 || mp2[ss]==1) can=1;
                eif(s.size()==3){
                    ss.erase(ss.end()-1, ss.end());
                    if (mp1[ss]) can = 1;
                }
                mp1[s]=1;
                if(s.size()==3){
                    s.erase(s.end()-1, s.end());
                    mp2[s]=1;
                }
            }
        }
    }
    if (can) cout << "YES" << endl;
    else cout << "NO" << endl;
}
int main() {
    int T;
    IOS;
    cin >> T;
    while (T--) {
        solve();
    }
}
