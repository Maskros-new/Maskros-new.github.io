ll quickpow(int a, int b){ 
    ll ans = 1;
    ll res = a;
    while (b)
    {
        if (b & 1)
            ans = ans * res;
        b >>= 1;
        res = res * res;
    }
    return ans;
}

{
    int n; cin>>n;

#include <iostream>
#include <algorithm>
#include <cstring>

int sum1[100005], sum2[100005];
int p[1000005];
int main() {
	int T;
	cin >> T;
	//T = 1;
	while(T--) {
		int n;
		cin >> n;
		string s;
		cin >> s;
		int ans = 0x3f3f3f3f;
		for(int i = 0; i < 26; i++) {
			string s1 = "";
			sum1[0] = sum2[n + 1] = sum2[n] = 0;
			char now = 'a' + i;
			for(int j = 0; j < s.size(); j++) {
				if(s[j] == now) {
					if(j == 0) sum1[j] = 1;
					else sum1[j] = sum1[j - 1] + 1;
				} else {
					if(j == 0) sum1[j] = 0;
					else sum1[j] = sum1[j - 1];
					s1 += s[j];
					p[s1.size() - 1] = j;
				}
			}

			for(int j = s.size() - 1; j >= 0; j--) {
				sum2[j] = sum2[j + 1];
				if(s[j] == now) sum2[j]++;
			}
			string s2 = s1;
			reverse(s2.begin(), s2.end());
			//cout << "fuck" << endl;
			if(s1 == s2) {
				//cout << (char)('a' + i) << endl;
				int del = 0;
				for(int k = 0; k < s1.size() / 2 + (s1.size() & 1 ? 1 : 0); k++) {
					//if(i == 0) cout << k << " " << endl;
					int pos1 = p[k], pos2 = p[s1.size() - 1 - k];
					//if(i + 'a' == 'r') cout << k << " " << del << endl;
					if(k == 0) del += abs(sum1[pos1] - sum2[pos2]);
					else del += abs((sum1[pos1] - sum1[p[k - 1]]) - (sum2[pos2] - sum2[p[s1.size() - k]]));
				}
				ans = min(ans, del);
			}
		}
		if(ans == 0x3f3f3f3f) cout << -1 << endl;
		else cout << ans << endl;
	}

	return 0;
}

D_Vupsen_Pupsen_and_0

题意：

已知非零数组a[i]，求非零数组b[i]，使得a[i]*b[i] 的和为0

思路：

个数为偶数两两相消，个数为奇数的话，选取尾部三个数，判断两个相加不为零的数出来，当作一个数与剩下的一个数两两相消即可。

坑点：三个数时合并的两个数相加不能为0

#include <bits/stdc++.h>

#define INF 0x3f3f3f3f3f
#define eps 1e-6
#define PI acos(-1.0)
#define pb push_back
#define eif else if
#define en putchar('\n')
#define rep(i, x, y) for (int i = x; i < y; i++)
#define red(i, x, y) for (int i = x; i > y; i--)
#define mem(a, x) memset(a, x, sizeof(a))
#define IOS cin.tie(0), ios::sync_with_stdio(false)
#define maxn 100005
typedef long long ll;
#define pll pair<ll, ll>
using namespace std;
void solve() {
    int a[maxn];
    int n;
    cin >> n;
    rep(i, 0, n) {
        cin >> a[i];
    }
    bool t1 = false, t2 = false, t3 = false;
    if (n & 1) {
        if (a[n - 1] + a[n - 2] != 0) {
            a[n - 2] = a[n - 1] + a[n - 2];
            t1 = true;
        } else {
            if (a[n - 2] + a[n - 3] != 0) {
                a[n - 3] = a[n - 2] + a[n - 3];
                a[n - 2] = a[n - 1];
                t2 = true;
            } else {
                a[n - 3] = a[n - 3] + a[n - 1];
                t3 = true;
            }
        }
        for (int i = 0; i < n - 1; i += 2) {
            if (i == n - 3) {
                if (t1) {
                    cout << a[i + 1] << " " << -a[i] << " " << -a[i] << " ";
                }
                eif(t2) {
                    cout << a[i + 1] << " " << a[i + 1] << " " << -a[i];
                }
                eif(t3) {
                    cout << a[i + 1] << " " << -a[i] << " " << a[i + 1];
                }
            } else {
                cout << a[i + 1] << " " << -a[i] << " ";
            }
        }
        cout << endl;
    } else {
        for (int i = 0; i < n; i += 2) {
            cout << a[i + 1] << " " << -a[i] << " ";
        }
        cout << endl;
    }
}
int main() {
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
}
