void solve() {
    int n; cin >> n;
    if(n % 7 == 0) cout << n << '\n';
    else{
        if(n >= 990) cout << 994 << '\n';
        else {
            n -= n%10;
            while(n % 7 != 0) n++;
            cout << n << '\n';
        }
    }
}

B_Minority

签

void solve() {
    string s; cin >> s;
    int one = 0, zero = 0, ans = 0;
    rep(i, 0, s.size()){
        if(s[i] == '0') zero ++; else one ++;
        ans = (one == zero)? ans :  min(one, zero); 
    }
    cout << ans << '\n';
}

C_Kill_ the_Monster

枚举k的分配种类即可

ceil(double x) ：double 类型向上取整

void solve() {
    double hc, dc, hm, dm, k, w, a;
    cin >> hc >> dc >> hm >> dm >> k >> w >> a;
    rep(i, 0, k+1){
        double hhc = 1.0 * i * a + hc;
        double ddc = 1.0 * (k - i) * w + dc;
        if(ceil(hm / ddc) <= ceil(hhc / dm)){
            cout << "YES\n";
            return ; 
        }
    }
    cout << "NO\n";
}

D_Make_Them_Equal (dp)

题意：

给定一个长为 nnn 的数组 aaa，初值全为 111 ，你可以做这样的操作：选择 iii 和 xxx (x>0) ，然后使 ai=ai+⌊aix⌋a_i = a_i + \lfloor\frac{a_i}{x}\rfloora​i​​=a​i​​+⌊​x​​a​i​​​​⌋

如果操作后 ai=bia_i = b_ia​i​​=b​i​​ ，你将得到 cic_ic​i​​ 的奖金，你需要在 kkk 次操作内使得奖金最大化，输出奖金的最大值。规定 1≤bi≤1e31 \leq b_i\leq 1e31≤b​i​​≤1e3 , 1≤ci≤1e61\leq c_i \leq 1e61≤c​i​​≤1e6

思路：

从 111 到 bib_ib​i​​ 的次数是固定的，根据 bib_ib​i​​ 的范围，我们可以打表预处理每个数的操作次数，然后考虑01背包，dp[i][j]表示前 i 个耗费为 j 的最大奖金，但是开1e6肯定会T，观察打表结果发现最大值为 12，这样可以优化一下dp的范围。

int b[1005], c[1005];
int pre[1005];
// 1 -> 1000 打表 max = 12
void init() {
    pre[1] = 0;
    rep(i, 1, 1001) {
        rep(j, 1, i + 1) {
            int tmp = i + i / j;
            if(tmp <= 1000) pre[tmp] = pre[tmp] ? min(pre[tmp], pre[i] + 1) : pre[i] + 1; 
        }
    }
}
int dp[12005];
void solve() {
    int n, k;
    cin >> n >> k;
    rep(i, 1, n + 1) cin >> b[i], b[i] = pre[b[i]];
    rep(i, 1, n + 1) cin >> c[i];
    int ans = 0;
    if(k >= 12 * n) {
        rep(i, 1, n + 1) ans += c[i]; 
    }else {
        rep(i, 0, 12 * n) dp[i] = 0;
        rep (i, 1, n + 1) {
            red(j, k + 1, 0){
                if(j - b[i] >= 0) dp[j] = max(dp[j], dp[j - b[i]] + c[i]);
                else break;
            }
        }
        ans = dp[k];
    }
    cout << ans << "\n";
}
