*A 九小时九个人九扇门 (dp (0/1背包))

题意：n 个人，每个人有一个数字 a[i]，可以随意组合，组合起来为数字的加和。一共1~9号门 ，如果组合后的数字根等于门的序号，即可打开。问分别有多少种人物组合可以打开对应的门。数字根：将该数字各数位上的数字相加得到一个新的数，直到得到的数字小于10为止。

有用的结论：一个数的数字根等于这个数对9取模的结果(得0时数字根为9)

思路：

经典0/1背包变形的DP问题，dp[i][j]表示考虑了前 i 个数，选择了一些数字使得求和对9取模得 j的方案数，转移类似0/1背包的转移

// 数字根 = 本身 mod 9
int a[maxn];
int dp[maxn][15];
void solve(){
    int n; cin>>n;
    int t;
    rep(i,1,n+1){
        cin>>t;
        a[i]=t%9;
    }
    dp[1][a[1]]=1;
    rep(i,2,n+1){
        rep(j,0,9){
            int k=(j+a[i])%9;
            dp[i][k]=(dp[i-1][k]%mod+dp[i-1][j]%mod)%mod;
        }
        dp[i][a[i]]++;
    }
    rep(i,1,10){
        if(i==9) cout<<dp[n][0];
        else cout<<dp[n][i]<<" ";
    }
}

C Baby’s first attempt on CPU

题意：计组学过的流水线CPU，解决先写后读相关问题需要中间插入空语句，共n个语句，每个语句三个数字表示和前1/2/3条语句是否存在先写后读相关，总共需要插入多少空语句

题解：水题，当时为了时间乱jb写的，无参考价值；维护三个dis表示和前1/2/3条语句的间隔条数，分类判断即可。

void solve() {
    int x1,x2,x3;
    int n; scanf("%d", &n);
    int ans=0;
    int dis1=0,dis2=1,dis3=2;
    rep(i,0,n){
        scanf("%d%d%d",&x1,&x2,&x3);
        if(x1==1) {ans+=3; dis2=4; dis3=5;}
        eif(x2==1) {
            if(dis2>=3){dis3=dis2+1; dis2=dis1+1; dis1=0; } 
            else {
                ans+=(3-dis2); dis2=4-dis2; dis3=3;
            }
        }eif(x3==1){
            if(dis3>=3) {dis3=dis2+1; dis2=dis1+1; dis1=0;}
            else{
                ans+=(3-dis3); dis2=4-dis3; dis3=4;
            }
        }else{
            dis3=dis2+1; dis2=dis1+1; dis1=0;
        }
    }
    cout<<ans<<endl;
}

D 牛牛做数论 (欧拉函数)

题意：

定义对于正整数 xxx, H(x)=ϕ(x)xH(x)=\dfrac{\phi(x)}{x}H(x)=​x​​ϕ(x)​​ ，给定一个整数 nnn，分别求 x0,x1∈[2,n]x_0,x_1 \in [2,n]x​0​​,x​1​​∈[2,n] 使得 H(x)H(x)H(x) 分别取最大值和最小值

题解：

① 根据欧拉函数的性质，我们知道如果 nnn 为质数，则 ϕ(n)=n−1\phi(n)=n-1ϕ(n)=n−1 ，此时 H(x)H(x)H(x)​ 即取最大值；

② 欧拉函数可以根据质因数分解求得：ϕ(x)=x∏p∣x(1−1p)\phi(x) = x\prod_{p|x}(1-\dfrac{1}{p})ϕ(x)=x∏​p∣x​​(1−​p​​1​​) , 回看 H(x)H(x)H(x)，发现可以约掉 xxx，故可知是由质数依次相乘一定是最小值: (1-\dfrac{1}{2})(1-\dfrac{1}{3})(1-\dfrac{1}{5})··· ，故可筛出质数后遍历质数表即可。

const int MAXN = 10000;
int prime[MAXN + 1];
void getPrime() {
    memset(prime, 0, sizeof(prime));
    for (int i = 2; i <= MAXN; i++) {
        if (!prime[i])
            prime[++prime[0]] = i;
        for (int j = 1; j <= prime[0] && prime[j] <= MAXN / i; j++) {
            prime[prime[j] * i] = 1;
            if (i % prime[j] == 0)
                break;
        }
    }
}
bool isPrime(int num) {
    if (num == 2 || num == 3) {
        return true;
    }
    if (num % 6 != 1 && num % 6 != 5) {
        return false;
    }
    for (int i = 5; i * i <= num; i += 6) {
        if (num % i == 0 || num % (i + 2) == 0) {
            return false;
        }
    }
    return true;
}
void solve() {
    int n;
    cin>>n;
    if(n==1) {cout<<-1<<endl; return;}
    int mx,mn=2;
    red(i,n+1,2){
        if(isPrime(i)) {mx=i; break;}
    }
    ll tmp=1;
    rep(i,1,1000){
        tmp*=prime[i];
        if(tmp<=n) mn=tmp;
        else break;
    }
    cout<<mn<<" "<<mx<<endl;
}

F 中位数切分

结论题：如果中位数大于 mmm，则该列中的大于 mmm 的数的个数一定超过一半，划分段数直接等于 大于m的个数 - 小于m的个数，随便推推就知道了。

void solve() {
    int n,m; scanf("%d%d",&n, &m);
    int bg=0,sm=0;
    int t;
    rep(i,0,n){
        scanf("%d", &t);
        if(t>=m) bg++;
        else sm++;
    } 
    if(bg<=n/2) cout<<-1<<endl;
    else cout<<bg-sm<<endl;
}

H 牛牛看云

题意：给出一个序列 aaa, 计算 ∑i=1n∑j=1n∣ai+aj−1000∣\sum_{i=1}^{n}\sum_{j=1}^{n}|a_i+a_j-1000|∑​i=1​n​​∑​j=1​n​​∣a​i​​+a​j​​−1000∣，数据范围 n<1e6, 0<=ai<=1000

题解：注意到 n<=1e6 且 a[i]<=1000，知道a[i]范围很小，所以极限数据下会有大量重复值出现，故可以用 cnt[i] 表示 i 出现的次数，枚举(i,j)对(共1e6种)：不同情况直接相乘，特殊处理相同情况

vector<int> v[maxn];
void solve() {
    int n;
    cin >> n;
    ll ans = 0;
    rep(i, 0, n) {
        int x;
        cin >> x;
        v[x].pb(i);
    }
    rep(i, 0, 1001) {
        rep(j, i, 1001) {
            ll x = v[i].size();
            ll y = v[j].size();
            if (i == j) {
                ans += x * (x + 1) / 2 * abs(i + j - 1000);
            } else
                ans += x * y * abs(i + j - 1000);
        }
    }
    cout << ans << endl;
}

*I B站与各唱各的 (概率/逆元)

题意：n个人唱m句，每个人对每一句都可以选择唱或不唱，n个人、m句之间互相独立，如果某一句没人唱或者所有人都唱了则失败，否则成功，求唱成功句子数量的期望值。对1e9+7取模。

题解：m×\dfrac{2^n-2}{2^n}​

由于句子间相互独立，则最终期望为一句话的期望乘 mmm；每个人唱第每句的唯一策略就是以 pip_ip​i​​ 的概率决定唱与不唱，所以失败的概率就为 ∏pi+∏(1−pi)\prod p_i+\prod (1-p_i)∏p​i​​+∏(1−p​i​​) （都唱+都不唱）

当 p1=p2=...=pn=12p_1=p_2=...=p_n=\dfrac{1}{2}p​1​​=p​2​​=...=p​n​​=​2​​1​​ 时，上式取最小值，故失败的概率为 22n\dfrac{2}{2^n}​2​n​​​​2​​ , 故易知成功的概率

注意求逆元, 推式子输出逆元的题比较常见。

ll quickpow(ll a, ll b) {
    ll ans = 1;
    ll res = a%mod;
    while (b) {
        if (b & 1)
            ans = ans * res%mod;
        b >>= 1;
        res = res * res%mod;
    }
    return ans%mod;
}
void solve() {
    int n,m; cin>>n>>m;
    ll a = (quickpow(2,n-1)-1+mod)%mod;
    a=a*m%mod;
    ll b = quickpow(2,n-1);
    ll ans = a*quickpow(b,mod-2)%mod;
    cout<<ans<<endl;
}

J 小朋友做游戏

水题

int a[maxn],b[maxn];
int ans = 0;
int posa,posb;
void goA(){
    ans+=a[posa];
    posa++;
}
void goB(){
    ans+=b[posb];
    posb++;
}
void solve() {
    int A,B,n; cin>>A>>B>>n;
    rep(i,0,A) cin>>a[i];
    rep(i,0,B) cin>>b[i]; 
    sort(a,a+A,greater<int>() );
    sort(b,b+B,greater<int>() );
    ans = 0;
    if(n>2*A) cout<<-1<<endl;
    else{
        posa=0,posb=0;
        if(n&1){
            rep(i,0,(n+1)/2) ans+=a[i]; posa=(n+1)/2;
        }else{
            rep(i,0,n/2) ans+=a[i]; posa=n/2;
        }
        bool AA=true,BB=true;
        if(posa==A) AA=false;
        if(posb==B) BB=false;
        rep(j,posa,n){
            if(AA&&BB){
                if (a[posa] >= b[posb]) {
                    goA();
                }else goB();
                if(posa==A) AA=false;
                if(posb==B) BB=false;
            }eif(AA){
                goA();
            }eif(BB) goB();
        }
        cout<<ans<<endl;
    }
}
