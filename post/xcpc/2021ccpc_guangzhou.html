H_Three Intergers (数学)

题意：给a,b,c，求满足x%y=a, y%z=b, z%x=c 的任意一组x,y,z

思路：数学题，方程化简，根据条件直接造，起初忘记了模的性质导致解不出方程，看了题解恍然大悟 题解

#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    int T;
    cin >> T;
    while (T--) {
        int a, b, c;
        cin >> a >> b >> c;
        if (a == b && b == c) {
            if (!a) {
                cout << "YES\n";
                cout << 1 << " " << 1 << " " << 1 << "\n";
            } else
                cout << "NO\n";
            continue;
        }
        ll x, y, z, k;
        if (b > a) {
            k = max(0, (c - a) / b) + 1;
            x = k * b + a;
            y = b;
            z = (k * b + a) * 2 + c;
        } else if (a > c) {
            k = max(0, (b - c) / a) + 1;
            x = a;
            y = (k * a + c) * 2 + b;
            z = k * a + c;
        } else if (c > b) {
            k = max(0, (a - b) / c) + 1;
            x = (k * c + b) * 2 + a;
            y = k * c + b;
            z = c;
        } else
            assert(false);
        cout << "YES\n";
        cout << x << " " << y << " " << z << "\n";
        assert(x % y == a && y % z == b && z % x == c);
        assert(x <= 1e18);
        assert(y <= 1e18);
        assert(z <= 1e18);
    }
    return 0;
}
