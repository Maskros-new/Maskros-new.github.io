void ekmp_pre(string x, ll next[])
{
    int m = x.length();
    next[0] = m;
    int j = 0;
    while (j + 1 < m && x[j] == x[j + 1])
        j++;
    next[1] = j;
    int k = 1;

    {

            k = i;
