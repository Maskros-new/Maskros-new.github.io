STL使用手册

Standard Template Library

又名《👴怕忘了先把他写下来》

我 是 谁 ?

斯丹德儿的 坦普雷特 莱博瑞 即标准模板库

  • 重要特点：数据结构和算法的分离

  • 重要特性：不是面向对象的

  • 逻辑层次：体现了泛型化程序设计的思想

  • 实现层次：以一种类型参数化的方式实现，基于模板(template)

六大组件：

  • 容器（Container），是一种数据结构，如list，vector，deques ，以模板类的方法提供 ;
  • 迭代器（Iterator），提供了访问容器中对象的方法。迭代器就如同一个指针。C++的指针也是一种迭代器。迭代器也可以是那些定义了operator*()以及其他类似于指针的操作符地方法的类对象；
  • 算法（Algorithm），是用来操作容器中的数据的模板函数。如sort()，find()，函数本身与他们操作的数据的结构和类型无关 ;
  • 仿函数（Functor）
  • 适配器（Adaptor）
  • 分配器（allocator）

👴 常用的

以下不按顺序，想到啥就写啥，有个🥚的顺序

容器

string

字符串

//构造函数
string s(str,index); //将字符串str内“始于位置index”的部分当作字符串的初值
string s(str,index,n); //将字符串str内“始于index且长度顶多n”的部分作为字符串的初值
string s(n,c); //生成一个字符串，包含n个c字符(char)
string s(str.begin(),str.end()); //以区间begin():end() (不包含end())内的字符作为字符串s的初值

//操作函数
s.erase(s.begin(),s.end()); //删除
s.replace(pos,len,str); //替换，从pos开始长为len被替换为str
s.find(str,pos=0); //从pos向右查找，返回str第一次出现的位置
s.rfind(str,pos=npos); //从pos反向查找，返回找到位置
s.find_first_of(str); //查找第一个属于str中字符的位置
s.find_first_not_of(str); //查找第一个不属于str中字符位置
s.find_last_of(str);
s.find_last_not_of(str);
s1.compare(s2); //比较，相等0大于1小于-1
s1.compare(pos,len,s2)//从s1的pos开始长为len与s2比较

stringstream

自动且直接的类型转换

stringstream sstream;
//数据类型转换
string str;
int a=123;
sstream << a;
sstream >> str;
cout<<str;
cout<<sstream.str(); 
//多个字符串拼接
sstream << "xxxx" << " " << "aaaaa";
//清空stringstream
sstream.clear();
sstream.str("");

pair

便携结构体

//访问通过first second
//可以通过一般运算符比较，先比较first再second
pair<string,int> p (str,123);
pair<string,int> p = make_pair(str,123);

vector

向量容器，动态存储

vector<int> v;
v.front(); v.back();
v.begin(); v.end();
v.push_back();
v.size();
v.empty();
v.clear();
v.pop_back(); //删除表尾元素
v.insert(it,x); //向迭代器it 指向的元素前插入新元素x
v.insert(it,n,x); //插入n个x
v.insert(it, first, last); //将由迭代器first和last 所指定的序列[first, last)插入到迭代器it指向的元素前面
v.erase(it); //删
v.erase(first,last);
v.reverse(n); //预分配缓冲空间，使存储空间至少可容纳n个元素
v.resize(n); //改变序列长度，超出元素删除，若源空间小于n，则默认值填满空间
v.resize(n,val); //超出的用val填满

stack

栈

stack<int> s;
s.push(x);
s.pop();
s.top();
s.empty();
s.size();

queue

队列，优先队列，双端队列

// queue
queue<int> q;
q.push(x);
q.pop();
q.front(); q.back();
q.empty();
q.size();

// priority_queue
priority<int> q; //默认大者优先
priority<int, vector<int>, greater<int> > q; //小的先出队 三个模板参数：元素类型，容器类型，比较算子
// 优先级设置：定义结构体
struct Student{
    int num;
    int grade;
    friend bool operator < (Student s1, Student s2){
        return s1.grade<s2.grade;
        //此处小于号还是小于的作用，故队列中以成绩高的学生优先，若反之变号即可
    }
};
priority_queue<Student> q;
//也可以把重载的函数写在结构体外面
struct cmp{
    bool operator() (Student s1, Student s2){
        return s1.grade<s2.grade;
    }
};//tips:优先队列中的cmp和sort中的cmp效果相反
priority_queue<Student,vector<Student>,cmp> q;
//如果结构体内数据庞大，像出现字符串或者数组，建议使用引用来提高效率。此时比较类的参数需要加上 const 和 &
friend bool operator < (const Student &s1, const Student &s2){
	return s1.grade<s2.grade;
}

//deque
deque<int> d;
d.push_back(x); //尾部插入
d.push_front(x); //首部插入元素
d.insert(d.begin()+len,x); //中间插入元素
//可以通过 d[0] d[1] 访问
d.pop_front(); //删除头部
d.pop_back();
d.erase(d.begin()+len); //删除指定位置元素，可以写多个

map

键值对，元素按关键字有序，内部红黑树

// map空间占用率高
map<string,int> m;
m[key]=value;
m.insert(make_pair(key,value));
m.erase(key);
m.erase(it);
m.clear();
m.size();	
map<string,int>::iterator it;
for(it=mp.begin();it!=mp.end();++it){
    cout<<it->first<<" "<<it->second<<endl;
}

// unordered_map 效率较高 内部哈希表
// 对于查找问题，unordered_map会更加高效一些，因此遇到查找问题，常会考虑一下用unordered_map

set

无重复，按键值自动排序

数据结构为红黑树。以节点形式保存（动态)，用迭代器访问

set<int> s;
s.begin(); s.end();
s.clear();
s.empty();
s.erase(x);
s.find(x);
s.insert(x);
s.size();
s.lower_bound(value); //返回第一个大于等于value的定位器
s.upper_bound(value); //返回最后一个大于等于value的定位器
set<int>::iterator it;
for(it=s.begin();it!=s.end();it++)
	cout<<*it<<" ";

list

链表

  • 如果你需要高效的随即存取，而不在乎插入和删除的效率，使用vector
  • 如果你需要大量的插入和删除，而不关心随即存取，则应使用list
  • 如果你需要随即存取，而且关心两端数据的插入和删除，则应使用deque
list<int> a(n);
list<int> a{1,2,3,4};
a.push_back(x);
a.push_front(x);
a.empty();
a.resize(n);
a.clear();
a.front(); a.back();
a.reverse();
a.merge(b); //b变空，a尾插入b
a.insert(pos,x);
a.insert(pos,b.begin(),b.end());
a.erase(a.begin(),a.end());
a.remove(x);

bitset

存放二进制0/1的容器，可以声明非常大的二进制位而不限于64，用来优化

bitset<N> b; //表示长为N的二进制串
// 通过b[pos]访问
b.any(); //存在为1的二进制位
b.none(); //不存在为1的二进制位
b.count(); //置1的二进制位的个数
b.size();
b.test(pos); //在pos位处是否为1
b.set(); //所有位置1
b.set(pos);
b.reset(); //所有位置0
b.reset(pos);
b.flip(); //逐位取反
b.flip(pos);

functions

查找

// 对于sort(a, a + n) 升序排列
lower_bound(a,a+n,x); //二分查找第一个大于等于x的位置
upper_bound(a,a+n,x); //二分查找第一个大于x的位置
// 对于sort(a, a + n, greater<int>()) 降序排列
lower_bound(a,a+n,x,greater<int>(); //二分查找第一个小于等于x的位置
upper_bound(a,a+n,x,greater<int>(); //二分查找第一个小于x的位置
            
find();
rfind();
