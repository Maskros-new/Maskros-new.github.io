  • 0x02 P1381 单词背诵 尺取

    尺取(双指针)： 对给定的一个序列，在序列中寻找包含全部需求的，长度最小的一段子序列 O(n)

    • 初始化左右下标 l, r 到适当位置
    • 不停向右移动r，直到 r 出界或者[l, r] 自区间已经满足要求
    • ans = min(ans, r - l + 1)
    • 移动 l 到适当位置，重复第一个过程
    #include<bits/stdc++.h>
using namespace std;

#define ll long long
#define int ll
#define p 233
#define mod 1610612741
#define maxn 1000005
int Hash1[1000005], Hash2[1000005], Hashtmp[1000005];
int b[1000005];
void add(int Hash[], string s) {
	Hash[0] = 0;
	for (int i = 0; i < s.size(); i++) {
		Hash[i + 1] = (Hash[i] * p + (s[i] - 'a')) % mod;
	}
}
void init() {
	b[0] = 1;
	for (int i = 1; i <= 1e6; i++) {
		b[i] = b[i - 1] * p % mod;
	}
}
int getHash(int Hash[], int l, int r) {
	return ((Hash[r] - Hash[l - 1] * b[r - l + 1]) % mod + mod) % mod;
}
map<ll, int> words;
map<ll, bool> vis;
string ss[maxn];
int cnt[maxn];
signed main() {
	init();
	int n; cin >> n;
	while (n--) {
		string s; cin >> s; add(Hash1, s);
		words[Hash1[s.size()]]++;
	}
	int m; cin >> m;
	int tot = 0;
	for (int i = 1; i <= m; i++) {
		cin >> ss[i];
		add(Hashtmp, ss[i]);
		Hash2[i] = Hashtmp[ss[i].size()];
		if (words[Hash2[i]] && !vis[Hash2[i]]) {
			vis[Hash2[i]] = 1; tot++; cnt[i] = tot;
		} else if (words[Hash2[i]]) {
			cnt[i] = tot;
		} else {
			cnt[i] = 0;
		}
	}
	cout << tot << '\n';
	if (!tot) cout << 0;
	else {
		int l = 1, r = 1;
		map<ll, int> haved;
		int ans = maxn;
		int tmp = 0;
		for (int l = 1, r = 1; r <= m; r++) {
			if (words[Hash2[r]]) {
				if (haved[Hash2[r]] == 0) {
					tmp++;
				}
				haved[Hash2[r]]++;
			}
			if (tmp < tot) continue;
			while (haved[Hash2[l]] > 1 || !words[Hash2[l]]) {
				if (haved[Hash2[l]] > 1) haved[Hash2[l]]--;
				l++;
			}
			ans = min(ans, r - l + 1);
		}

		cout << ans;
	}

}
